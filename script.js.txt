const texts = {
  easy: "This is an easy sentence.",
  medium: "Typing practice improves your speed and accuracy.",
  hard: "Consistency and discipline are the keys to mastering any skill."
};

let time = 0;
let mistakes = 0;

const textEl = document.getElementById("text");
const input = document.getElementById("input");
const level = document.getElementById("level");

function loadText() {
  textEl.innerText = texts[level.value];
}
loadText();

level.addEventListener("change", loadText);

input.addEventListener("input", () => {
  let original = textEl.innerText;
  let typed = input.value;

  mistakes = 0;

  for (let i = 0; i < typed.length; i++) {
    if (typed[i] !== original[i]) {
      mistakes++;
    }
  }

  document.getElementById("mistakes").innerText = mistakes;
});

setInterval(() => {
  time++;
  document.getElementById("time").innerText = time;
}, 1000);
